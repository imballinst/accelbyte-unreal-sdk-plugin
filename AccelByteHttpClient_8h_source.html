678 FString ParseUStructToJsonString(TSharedPtr<FJsonObject> JsonObject, bool bOmitBlankValues = false)
679 {
680 FString JsonString;
681
682 if (!JsonObject.IsValid())
683 {
684 UE_LOG(LogTemp, Error, TEXT("HttpClient Request UStructToJsonObject failed!"));
685
686 return JsonString;
687 }
688
689 // Omit blank JSON string values if opted into by the caller
690 if (bOmitBlankValues)
691 {
692 FAccelByteUtilities::RemoveEmptyStrings(JsonObject);
693 }
694
695 // Finally, write the JSON object to a string
696 TSharedRef<TJsonWriter<>> const Writer = TJsonWriterFactory<>::Create(&JsonString);
697 if (!FJsonSerializer::Serialize(JsonObject.ToSharedRef(), Writer))
698 {
699 UE_LOG(LogTemp, Error, TEXT("HttpClient Request FJsonSerializer::Serialize failed!"));
700
701 return JsonString;
702 }
703
704 return JsonString;
705 }

707 template<typename U, typename V>
708 FAccelByteTaskPtr ProcessRequest(FHttpRequestPtr& Request
709 , FString const& Verb
710 , FString const& Url
711 , FHttpFormData const& QueryParams
712 , TMap<FString, FString> const& Headers
713 , U const& OnSuccess
714 , V const& OnError)
715 {
716 if (!IsValidUrl(Url))
717 {
718 FString ErrorText = FString::Format(TEXT("Invalid URL format {0}"), { Url });
719 ExecuteError(OnError, ErrorText);
720 return nullptr;
721 }
722
723 FString QueryString;
724
725 if (!QueryParams.IsEmpty())
726 {
727 QueryString.Append("?").Append(QueryParams.ToUrlEncodedString());
728 }
729
730 Request->SetURL(FString::Printf(TEXT("%s%s"), *Url, *QueryString));
731 Request->SetVerb(Verb.ToUpper());
732
733 for (const auto& Kvp : Headers)
734 {
735 Request->SetHeader(Kvp.Key, Kvp.Value); // Override existing headers
736 }
737
738 return HttpRef.ProcessRequest(Request
739 , CreateHttpResultHandler(
740 OnSuccess,
741 OnError,
742 &HttpRef
743 )
744 , FPlatformTime::Seconds());
745 }
