81 inline void FreeCache() override
82 {
83 FDirectoryPath Directory = DataStorage.GetAbsoluteFileDirectory();
84 FString AbsoluteDirPath = Directory.Path;
85 TArray<FString> Files;
86 IFileManager::Get().FindFiles(Files, *AbsoluteDirPath, *EXTENSION_CACHE_NAME);
87
88 for (int i = 0; i < Files.Num(); i++)
89 {

103 inline void RemoveCache(const FName& Key) override
104 {
105 auto Filename = ConvertKeyToFilename(Key);
106 size_t CurrentSize = this->ArrayGetIndex(this->FindIndex(Key)).Length;
107
108 IFileManager::Get().Delete(*Filename, true, true, true);
109 CurrentFileCount -= 1;
110 CurrentFileSizeBytes -= CurrentSize;
111
112 for (int i = 0; i < DerivedChunkArray.Num(); i++)
113 {
114 if (DerivedChunkArray[i].Key == Key)

122 inline bool FreeCacheBeforeInsertion(T& Item) override
123 {
124 auto Required = FAccelByteLRUCache<T>::GetRequiredSize(Item);
125 size_t Left = MaxFileSizeBytes - CurrentFileSizeBytes;
126
127 bool bChunkCountIsSafe = CurrentFileCount < MaxFileCount;
128
129 if (Left >= Required && bChunkCountIsSafe)
130 {
131 return true;
132 }
133
134 auto Node = this->DLLGetTail();
135 while (Node != nullptr && (Required > Left || this->ArrayGetNum() >= MaxFileCount))
136 {
137 size_t TailSize = Node->GetValue().Length;
138 this->Remove(this->DLLGetTail()->GetValue().Key);
139 Left += TailSize;
140 Node = this->DLLGetTail();
141 }
142
143 size_t ModifiedStorageSizeLeft = MaxFileSizeBytes - CurrentFileSizeBytes;
144 return (Required <= ModifiedStorageSizeLeft) && (this->ArrayGetNum() < MaxFileCount);
145 }

147 inline const FAccelByteCacheWrapper<T>* InsertToCache(T& Item, const FName& Key) override
148 {
149 if (!InsertPrerequisiteOkay())
150 {
151 return nullptr;
152 }
153
154 auto Size = FAccelByteLRUCache<T>::GetRequiredSize(Item);
155 FAccelByteCacheWrapper<T> Result{ Key, nullptr, Size};
156
157 TArray<uint8> ArrayByte = ToArrayByte(Item);
158 if (ArrayByte.Num() == 0)
159 {
160 return nullptr;
161 }
162
163 auto AbsPath = CompleteFilenameToAbsolute(ConvertKeyToFilename(Key));
164 bool bIsSuccess = FFileHelper::SaveArrayToFile(ArrayByte, *AbsPath);
165
166 if (!bIsSuccess) return nullptr;
167 int Index = DerivedChunkArray.Add(Result);
168 CurrentFileCount += 1;
169 CurrentFileSizeBytes += Result.Length;
170
171 return &DerivedChunkArray[Index];
172 }
173
174 inline bool InsertPrerequisiteOkay()
175 {
176 if (CurrentFileCount >= MaxFileCount)
177 {
178 return false;
179 }
180 if (CurrentFileSizeBytes >= this->MaxFileSizeBytes)
181 {
182 return false;
183 }
184 return true;
185 }

190 inline TSharedPtr<T> GetTheValueFromChunkArray(int Index) override
191 {
192 FAccelByteCacheWrapper<T>& ChunkInfo = this->ArrayGetIndex(Index);
193 FString Key = ChunkInfo.Key.ToString();
194
195 TArray<uint8> ArrayByte;
196 auto AbsPath = CompleteFilenameToAbsolute(ConvertKeyToFilename(Key));
197 bool bIsOK = FFileHelper::LoadFileToArray(ArrayByte, *AbsPath);
198 if (!bIsOK || ArrayByte.Num() == 0) return nullptr;
199
200 FString StringValue = FAccelByteArrayByteFStringConverter::BytesToFString(ArrayByte, false);
201
202 TSharedPtr<T> Output = FromFString(StringValue);
203 if (Output.IsValid())
204 {
205 ChunkInfo.Data = Output;
206 return ChunkInfo.Data;
207 }
208 else
209 {
210 return nullptr;
211 }

269 auto HeaderSplitter = [this](const TArray<FString>& HeaderAndValuePair, TDelegate<void(FString /*Header*/, FString /*Value*/)> ActionSetHeader)
270 {
271 for (int i = 0 ; i < HeaderAndValuePair.Num(); i++)
272 {
273 const FString& HeaderAndValue = HeaderAndValuePair[i];
274 FString Header, Value;
275 HeaderAndValue.Split(":", &Header, &Value);
276
277 Header = Header.TrimStartAndEnd();
278 Value = Value.TrimStartAndEnd();
279
280 ActionSetHeader.Execute(Header, Value);
281 }
282 };
283
284 FHttpRequestPtr Request = MakeShareable<IHttpRequest>(FGenericPlatformHttp::ConstructRequest());
285 // Cannot set response header at all
286 FHttpResponsePtr Response = nullptr;
287
288 Request->SetURL(SerializeableHttpStruct.RequestURL);
289 HeaderSplitter(SerializeableHttpStruct.RequestHeaders, TDelegate<void(FString, FString)>::CreateLambda([&](FString Header, FString Value) { Request->SetHeader(Header, Value); }));
290
291 TSharedPtr<FAccelByteHttpCacheItem> Output = MakeShareable<FAccelByteHttpCacheItem>(new FAccelByteHttpCacheItem);
292 Output->SerializableRequestAndResponse = SerializeableHttpStruct;
293 Output->Request = Request;
