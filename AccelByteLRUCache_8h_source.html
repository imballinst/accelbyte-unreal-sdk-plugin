155 inline bool Emplace(const FName& Key, T& Item)
156 {
158 Remove(Key);
159
160 if (FreeCacheBeforeInsertion(Item) == false)
161 {
162 return false;
163 }
164
165 // After the existing removed, it can be added to the front
166 const FAccelByteCacheWrapper<T>* Result = InsertToCache(Item, Key);
167 if (Result == nullptr)
168 {
169 return false;
170 }
171 ArrayAdd(Result);
172 DLLAddHead(*Result);

184 inline TSharedPtr<T> Find(const FName& Key, bool bPeekOnly = false)
