14template<typename T>
15class DBQueryTask : public FNonAbandonableTask
16{
17 friend class FAutoDeleteAsyncTask<DBQueryTask>;
18 const TSharedPtr<T> Result;
19 const AccelByte::THandler<T> OnDone;
20 const double DelayTime;
21 const double TotalTimeOut;
22
23 DBQueryTask(const TSharedPtr<T>& Result, const AccelByte::THandler<T>& OnDone, double Delay = 5.0, double TimeOut = 60.0)
24 : Result(Result)
25 , OnDone(OnDone)
26 , DelayTime(Delay)
27 , TotalTimeOut(TimeOut)
28 {}
29
30 void DoWork()
31 {
32 const double TaskInitialTime = FPlatformTime::Seconds();
33
34 while (!Result.IsValid() && ((FPlatformTime::Seconds() - TaskInitialTime) < TotalTimeOut))
35 {
36 FPlatformProcess::Sleep(DelayTime);
37 UE_LOG(LogTemp, Warning, TEXT("Result: %d"), Result.IsValid());
38 }
39
40 OnDone.ExecuteIfBound(*Result.Get());
41 }
42 FORCEINLINE TStatId GetStatId() const { RETURN_QUICK_DECLARE_CYCLE_STAT(DBQueryTask, STATGROUP_ThreadPoolAsyncTasks); }
43
44};

229 inline void FAccelByteSQLite3::SaveItemToTable(const FString& TableName, const FString& Key, const T& DataToInsert, const THandler<bool>& OnDone)
230 {
231 CreateTable(TableName, DataToInsert.StaticStruct(), THandler<bool>::CreateLambda([this, TableName, Key, DataToInsert, OnDone](bool CreateResult)
232 {
233 if (CreateResult)
234 {
235 FAutoDeleteAsyncTask<DBQueryTask<SQLiteQueryResult>>* MyTask = new FAutoDeleteAsyncTask<DBQueryTask<SQLiteQueryResult>>(MakeShareable(USQLiteDatabase::RunQueryAndGetResults(
236 DatabaseName,
237 FString::Printf(TEXT("Select name, type FROM PRAGMA_table_info('%s') ORDER BY cid;"), *TableName)).Release()),
238 THandler<SQLiteQueryResult>::CreateLambda([this, Key, DataToInsert, TableName, OnDone](SQLiteQueryResult Result)
239 {
240
241 FSQLiteTableRowSimulator Row;
242 TSharedPtr<FJsonObject> JsonObject = FJsonObjectConverter::UStructToJsonObject(DataToInsert);
243 bool bIsSuccess = true;
244 for (auto Column : Result.Results)
245 {
246 FSQLiteTableField Field = JsonObjectToTableField(JsonObject, Column.Fields[0].StringValue, Column.Fields[1].StringValue);
247 if (Field.FieldName.IsEmpty())
248 {
249 bIsSuccess = false;
250 break;
251 }
252 if (Field.FieldName == TEXT("'Key'"))
253 {
254 Field.FieldValue = FString::Printf(TEXT("\"%s\""), *Key);
255 }
256 Row.rowsOfFields.Add(Field);
257 }
258 if (bIsSuccess)
259 {
260 USQLiteDatabase::InsertRowsIntoTableUpsert(DatabaseName, TableName, TArray<FSQLiteTableRowSimulator>{ { Row }}, TEXT("Key"));
261 }
262 OnDone.ExecuteIfBound(bIsSuccess);
263 }));
264 MyTask->StartBackgroundTask();
265 }
266 else
267 {
268 OnDone.ExecuteIfBound(CreateResult);
269 }
270 }));
271 }

273 template<typename T>
274 inline void FAccelByteSQLite3::GetItemsFromTable(const FString& TableName, const THandler<TMap<FString, T>>& OnDone, int Limit, int Offset)
275 {
276 FSQLiteDatabaseReference DBRef{ DatabaseName, {TableName} };
277
278 FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>* MyTask = new FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>(MakeShared<FSQLiteQueryResult>(USQLiteDatabase::GetDataBP(
279 DBRef,
280 { "*" },
281 FSQLiteQueryFinalizedQuery{},
282 Limit,
283 Offset)),
284 THandler<FSQLiteQueryResult>::CreateLambda([this, OnDone](FSQLiteQueryResult Result)
285 {
286 if (Result.Success)
287 {
288 TMap<FString, T> Data;
289 for (int i = 0; i < Result.ResultRows.Num(); i++)
290 {
291 FString Key;
292 FString JsonString = TEXT("{");
293 bool bIsNotFirstField = false;
294 for (int j = 0; j < Result.ResultRows[i].Fields.Num(); j++)
295 {
296 if (Result.ResultRows[i].Fields[j].Key == TEXT("Key"))
297 {
298 Key = Result.ResultRows[i].Fields[j].Value;
299 Key.TrimQuotesInline();
300 continue;
301 }
302 if (bIsNotFirstField)
303 {
304 JsonString += TEXT(", ");
305 }
306 else
307 {
308 bIsNotFirstField = true;
309 }
310 JsonString += FString::Printf(TEXT("\"%s\" : %s"), *Result.ResultRows[i].Fields[j].Key, *Result.ResultRows[i].Fields[j].Value);
311 }
312 JsonString += TEXT("}");
313 typename std::remove_const<typename std::remove_reference<T>::type>::type UStructData;
314 FJsonObjectConverter::JsonObjectStringToUStruct(JsonString, &UStructData, 0, 0);
315 Data.Add(Key, UStructData);
316 }
317 OnDone.ExecuteIfBound(Data);
318 }
319 else
320 {
321 OnDone.ExecuteIfBound(TMap<FString, T>{});
322 }
323 }));
324 MyTask->StartBackgroundTask();
325 }

328 inline void FAccelByteSQLite3::GetItemFromTable(const FString& TableName, const FString& Key, const THandler<TPair<FString, T>>& OnDone)
329 {
330 FSQLiteDatabaseReference DBRef{ DatabaseName, {TableName} };
331 const FString Condition = FString::Printf(TEXT("Key = '\"%s\"'"), *Key);
332
333 FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>* MyTask = new FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>(MakeShared<FSQLiteQueryResult>(USQLiteDatabase::GetDataBP(
334 DBRef,
335 { "*" },
336 FSQLiteQueryFinalizedQuery{ Condition })),
337 THandler<FSQLiteQueryResult>::CreateLambda([this, Key, OnDone](FSQLiteQueryResult Result)
338 {
339 if (Result.Success)
340 {
341 TPair<FString, T> Data{};
342 for (int i = 0; i < Result.ResultRows.Num(); i++)
343 {
344 FString JsonString = TEXT("{");
345 bool bIsNotFirstField = false;
346 bool bKeyFound = false;
347 for (int j = 0; j < Result.ResultRows[i].Fields.Num(); j++)
348 {
349 if (Result.ResultRows[i].Fields[j].Key == TEXT("Key"))
350 {
351 if (Result.ResultRows[i].Fields[j].Value.TrimQuotes() == Key)
352 {
353 bKeyFound = true;
354 }
355 else
356 {
357 break;
358 }
359 continue;
360 }
361 if (bIsNotFirstField)
362 {
363 JsonString += TEXT(", ");
364 }
365 else
366 {
367 bIsNotFirstField = true;
368 }
369 JsonString += FString::Printf(TEXT("\"%s\" : %s"), *Result.ResultRows[i].Fields[j].Key, *Result.ResultRows[i].Fields[j].Value);
370 }
371 JsonString += TEXT("}");
372 if (bKeyFound)
373 {
374 typename std::remove_const<typename std::remove_reference<T>::type>::type UStructData;
375 FJsonObjectConverter::JsonObjectStringToUStruct(JsonString, &UStructData, 0, 0);
376 Data = TPair<FString, T>{ Key, UStructData };
377 break;
378 }
379 }
380 OnDone.ExecuteIfBound(Data);
381 }
382 else
383 {
384 OnDone.ExecuteIfBound(TPair<FString, T>{});
385 }
386 }));
387 MyTask->StartBackgroundTask();
388 }
