167 template<typename CharType = TCHAR, template<typename> class PrintPolicy = TPrettyJsonPrintPolicy, typename InStructType>
168 static bool TArrayUStructToJsonString(TArray<InStructType> const& InArray
169 , FString& OutJsonString
170 , int64 CheckFlags = 0
171 , int64 SkipFlags = 0
172 , int32 Indent = 0)
173 {
174 const UStruct* StructDefinition = InStructType::StaticStruct();
175 TArray< TSharedPtr<FJsonValue> > JsonArray;
176
177 for (auto& Item : InArray)
178 {
179 const void* Struct = &Item;
180 TSharedPtr<FJsonObject> JsonObject = MakeShareable(new FJsonObject());
181 if (FJsonObjectConverter::UStructToJsonObject(StructDefinition, Struct, JsonObject.ToSharedRef(), CheckFlags, SkipFlags, nullptr))
182 {
183 TSharedRef< FJsonValueObject > JsonValue = MakeShareable(new FJsonValueObject(JsonObject));
184 JsonArray.Add(JsonValue);
185 }
186 }
187
188 TSharedRef<TJsonWriter<CharType, PrintPolicy<CharType>>> JsonWriter = TJsonWriterFactory<CharType, PrintPolicy<CharType>>::Create(&OutJsonString, Indent);
189 if (FJsonSerializer::Serialize(JsonArray, JsonWriter))
190 {
191 JsonWriter->Close();
192 return true;
193 }
194 else
195 {
196 UE_LOG(LogJson, Warning, TEXT("UStructToFormattedObjectString - Unable to write out json"));
197 JsonWriter->Close();
198 }
199
200 return false;
201 }

292 template<typename ObjectType>
293 static bool UStructArrayToJsonObjectString(TArray<ObjectType> const& Objects
294 , FString& OutString)
295 {
296 OutString.Append(TEXT("["));
297 FString JsonArrayString = TEXT("");
298 for (auto& Item : Objects)
299 {
300 FString Delimiter = JsonArrayString.IsEmpty() ? TEXT("") : TEXT(",");
301 JsonArrayString.Append(Delimiter);
302 FString JsonObjectString = TEXT("");
303 if (!FJsonObjectConverter::UStructToJsonObjectString(Item, JsonObjectString))
304 {
305 return false;
306 }
307 JsonArrayString.Append(JsonObjectString);
308 }
309 OutString.Append(JsonArrayString);
310 OutString.Append(TEXT("]"));
311 return true;
312 }
